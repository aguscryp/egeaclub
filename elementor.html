function draw(obj) { const group = new THREE.Group(); group.position.set(obj.pos.x, 0, obj.pos.z); if (obj.type === 'calle') { const geo = new THREE.BoxGeometry(50, 0.4, 50); const mat = new THREE.MeshStandardMaterial({ color: 0x222222 }); group.add(new THREE.Mesh(geo, mat)); } else if (obj.type === 'sw4') { // CUERPO DE LA CAMIONETA const body = new THREE.Mesh(new THREE.BoxGeometry(8, 4, 16), new THREE.MeshStandardMaterial({color: 0xeeeeee})); body.position.y = 3; const cabina = new THREE.Mesh(new THREE.BoxGeometry(7, 3, 8), new THREE.MeshStandardMaterial({color: 0x333333})); cabina.position.set(0, 6, -1); // RUEDAS const rGeo = new THREE.CylinderGeometry(1.2, 1.2, 1, 12); const rMat = new THREE.MeshStandardMaterial({color: 0x111111}); for(let i=0; i<4; i++) { const r = new THREE.Mesh(rGeo, rMat); r.rotation.z = Math.PI/2; r.position.set(i%2?4:-4, 1.2, i<2?5:-5); group.add(r); } group.add(body, cabina); } else if (obj.type === 'estacion') { // PLAYÓN const playon = new THREE.Mesh(new THREE.BoxGeometry(40, 0.5, 30), new THREE.MeshStandardMaterial({color: 0x444444})); // TECHO IMPONENTE const techo = new THREE.Mesh(new THREE.BoxGeometry(42, 2, 32), new THREE.MeshStandardMaterial({color: 0x00ff44})); techo.position.y = 15; // COLUMNAS const colGeo = new THREE.CylinderGeometry(1, 1, 15); const colMat = new THREE.MeshStandardMaterial({color: 0xcccccc}); for(let i=0; i<2; i++) { const c = new THREE.Mesh(colGeo, colMat); c.position.set(i%2?15:-15, 7.5, 0); group.add(c); } group.add(playon, techo); } else if (obj.type === 'farola') { const poste = new THREE.Mesh(new THREE.CylinderGeometry(0.3, 0.5, 20), new THREE.MeshStandardMaterial({color: 0x333333})); poste.position.y = 10; const lampara = new THREE.Mesh(new THREE.SphereGeometry(1.5), new THREE.MeshBasicMaterial({color: 0xffff00})); lampara.position.y = 20; group.add(poste, lampara); } else if (obj.type === 'taller') { // EDIFICIO AGUSLAVADOS const base = new THREE.Mesh(new THREE.BoxGeometry(25, 12, 25), new THREE.MeshStandardMaterial({color: 0x00d4ff, transparent: true, opacity: 0.8})); base.position.y = 6; const logo = new THREE.Mesh(new THREE.BoxGeometry(10, 4, 1), new THREE.MeshStandardMaterial({color: 0xff007f})); logo.position.set(0, 10, 12.6); group.add(base, logo); } else if (obj.type === 'pino') { const tronco = new THREE.Mesh(new THREE.CylinderGeometry(0.8, 1, 5), new THREE.MeshStandardMaterial({color: 0x3d2b1f})); tronco.position.y = 2.5; const hojas = new THREE.Mesh(new THREE.ConeGeometry(6, 15, 8), new THREE.MeshStandardMaterial({color: 0x1a4d1a})); hojas.position.y = 12.5; group.add(tronco, hojas); } else { // Default para otros objetos const geo = new THREE.BoxGeometry(10, 10, 10); const mat = new THREE.MeshStandardMaterial({color: 0xff007f}); const mesh = new THREE.Mesh(geo, mat); mesh.position.y = 5; group.add(mesh); } scene.add(group); }
